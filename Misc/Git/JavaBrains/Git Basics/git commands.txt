git commit --amend

git log --name-status

git checkout -- .

git checkout -b <branch-name> # create the branch and switch to it

git restore --staged <filename>

git branch -d <branch-name> # delete branch
git branch -a # show all branches (incl. remote)

git rebase <branch-name>
